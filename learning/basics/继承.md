```js
var obj = {}; obj.a = 100;
var arr = []; arr.a = 100;
function fn1() { }
fn1.a = 100;

// 要点二：__proto__
console.log(obj.__proto__); //Object.prototype
console.log(arr.__proto__); //Array.prototype
console.log(fn1.__proto__); //Function.prototype

// 要点三：函数有 prototype
console.log(fn1.prototype); //{constructor: fn1()}
```
